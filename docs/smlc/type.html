575class _Compose(ValueConstraint):
576    """Compose value constraints. This is a workaround to implement length constraint
577    for strings. We can override `is_valid` semantics to return int, not bool. Then we
578    can pass this int to other value constraints and eventually get bool.
579
580    >>> r = _Compose(Equal(2), _Length())
581    >>> r("a")
582    False
583    >>> r("aa")
584    True
585    >>> r("aaa")
586    False
587    """
588
589    def __init__(self, *args, backward: bool = True) -> None:
590        self._backward = backward
591        if len(args) > 1:
592            # args is a list of constraints
593            self._constraints = args
594        elif len(args) == 1:
595            if isinstance(args[0], list):
596                # args[0] is a list of constraints
597                self._constraints = args[0]
598            else:
599                # args[0] is the only constraint
600                self._constraints = [args[0]]
601        else:
602            # len(args) == 0
603            self._constraints = []
604
605    def is_valid(self, value) -> bool:
606        if not self._backward:
607            for r in self._constraints:
608                value = r.is_valid(value)
609        else:
610            for r in reversed(self._constraints):
611                value = r.is_valid(value)
612        return value
